Stack VM T01
Commands
        PUSH = 0x10, Загрузка в стек из озу по адресу из стека
        POP = 0x11, Запись из стека в озу по адресу из стека
        PNC = 0x12, Загрузка в стек из озу по адресу (текущий адрес + 1) 
        ADD = 0x20, Запись в стек суммы двух последних значений
        SUB = 0x21,  Запись в стек разности двух последних значений
        READ = 0x30, Запись в стек кода введённого символа
        WRITE = 0x31, Запись в терминал символа по коду из стека
        STORE = 0x40, Запись на диск значения из стека. (Конец стека, значение, страница, ячейка) 
        LOAD = 0x41, Чтение в стек значения из диска. (Конец стека, страница, ячейка) 
        CMP = 0x50, Сравниваем два последних значения из стека и устанавливаем флаги ZeroFlag если они равны и NegativeFlag, если значение 2 (Дальнее от конца стека) больше значения 1
        JZ = 0x60, Переход по значению из стека, если стоит ZeroFlag
        JLS = 0x61, Переход по значению из стека, если стоит NegativeFlag
        JMP = 0x60, Переход по значению из стека
        HLT = 0xff, Конец программы.
